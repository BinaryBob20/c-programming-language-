#include <stdio.h>
#include <math.h>

int main() {
    float x, y, r, theta;

    printf("Enter Cartesian coordinates (x, y): ");
    scanf("%f %f", &x, &y);
 
    r = sqrt(x * x + y * y);

    theta = atan(y / x);
    
    printf("Polar coordinates (r, Î¸): (%.f, %.f radians)\n", r, theta);
    
    return 0;
}
